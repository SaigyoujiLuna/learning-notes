珂朵莉树（Chtholly Tree），又名老司机树 ODT（Old Driver Tree）．起源自 CF896C．

这个名称指代的是一种「使用平衡树（std::set、std::map 等）或链表（std::list、手写链表等）维护颜色段均摊」的技巧，而不是一种特定的数据结构．其核心思想是将值相同的一段区间合并成一个结点处理．相较于传统的线段树等数据结构，对于含有区间覆盖的操作的问题，珂朵莉树可以更加方便地维护每个被覆盖区间的值
# 实现代码
```c
typedef long long ll;
typedef struct {
    int l, r;
    ll v;
} Seg;
typedef struct {
    Seg *a;
	int sz, cap;
} ODT;

// C语言辅助内存管理
static void odt_reserve(ODT *t, int need) {
    if (t->sz + need <= t->cap) return;
    while (t->cap < t->sz + need) t->cap *= 2;
	t->a = realloc(t->a, t->cap * sizeof(Seg));
}

static int odt_lb(ODT *t, int pos) {
    int lo = 0, hi = t->sz - 1, ans = -1;
    while (lo <= hi) {
        int mid = ((hi - lo) >> 1) + lo; 
		if (t->a[mid].l <= pos) {
		    ans = mid;
		    lo = mid + 1
		} else {
		    hi = mid - 1;
		}
    }
	return ans;
}

ODT odt_new(int n, ll val) {
    ODT t;
    t.cap = 16;
    t.sz = 1;
    t.a = malloc(t.cap * sizeof(Seg));
    t.a[0] = (Seg) {1, n, val};
	return t;
}

int odt_split(ODT *t, int pos) {
	int i = odt_lb(t, pos);
	if (i >= 0 && t->a[i].l == pos) {
		return i;
	}
	int r = t->a[i].r;
	ll v = t->a[i].v;
	t->a[i].r = pos - 1;
	odt_reserve(t, 1);
	memmove(t->a + i + 2, t->a + i + 1, (t->sz - i - 1) * sizeof(Seg));
	t->a[i + 1] = (Seg){pos, r, v};
	t->sz++;
	return i + 1;
}

void odt_split2(ODT *t, int l, int r, int *il, int *ir) {
    int saved_ir = odt_split(t, r + 1);
    int sz_before = t->sz;
    *il = odt_split(t, l);
    *ir = saved_ir + (t->sz - sz_before);
}

 void odt_assign(ODT *t, int l, int r, ll v) {
     int il, ir;
     odt_split2(t, l, r, &il, &ir);
	 t->a[il] = (Seg) {l, r, v};
	 int del = ir - il - 1;
	 if (del > 0) {
	     memmove(t->a + il + 1, t->a + ir, (t->sz - ir) * sizeof(Seg));
		 t->sz -= del;
	 }
 }

 void odt_add(ODT *t, int l, int r, ll val) {
     int il, ir;
     odt_split2(t, l, r, &il, &ir);
	 for (int i = il; i < ir; i++) {
	     t->a[i].v += val;
	 }
 }
 
 ll odt_sum(ODT *t, int l, int r) {
     int il, ir;
     odt_split2(t, l, r, &il, &ir);
	 ll s = 0;
	 for (int i = il; i < ir; i++) {
	     s += (ll)(t->a[i].r - t->a[i].l + 1) * t->a[i].v;
	 }
	 return s;
 }
```
