C++11新增的嵌套命名空间特性，允许将子空间的函数和类型导出到父命名空间中
```cpp
#include <iostream>
namespace Parent {
	namespace Child1 
	{
		void foo() {std::cout << "Child1::foo()" << std::endl; }
	}
	// 内联命名空间的所有函数会导出到父空间
	inline namespace Child2
	{
		void foo() {std::cout << "Child2::foo()" << std::endl; }
	}
}


Parent::Child1::foo();
Parent::foo();

namespace A::B::inline C{
	int foo() {return 5;}
}

A::B::foo();
```
```output
Child1::foo(Child2::foo()
)
```
